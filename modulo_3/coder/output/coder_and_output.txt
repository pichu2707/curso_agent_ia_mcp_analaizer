def calculate_series(n):
    total = 0.0
    for i in range(n):
        term = (-1) ** i / (2 * i + 1)
        total += term
    return total * 4

# Calcular los primeros 10000 términos
total_sum = calculate_series(10000)

# Imprimir resultado
print(total_sum)
